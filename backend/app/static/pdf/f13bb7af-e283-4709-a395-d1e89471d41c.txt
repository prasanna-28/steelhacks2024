\section{Set Operations}
\begin{align*}
A \cup B &= \{c|c\in A \text{ or } c\in B\} \text{ *union*} \\
A\cap B &= \{c|c\in A \text{ and } c\in B\} \text{ *intersection*}\\
A\backslash B &= \{c\in A| c\notin B\} \text{ *difference*}\\
A \Delta B &= A\cup B \backslash (A \cap B) \text{ *symmetric difference*}
\end{align*}
\section{Families of Sets}
Let $(A_\alpha)_{\alpha \in I}$ be an *indexed family of sets* (index set I).
\begin{align*}
\text{EX: } (1) \ \  I &= \mathbb{N}\\
A_\alpha &:= \{\alpha, \alpha + 1\}\\
(2) \ \  I &= \mathbb{R}^+ = \{x\in\mathbb{R}| x > 0\} \\
A_\alpha &:= [\alpha, \infty)
\end{align*}
\begin{align*}
\text{union: } \bigcup_{\alpha \in I} A_\alpha &= \{\alpha| \alpha \in A_\alpha \text{ for some } \alpha \in I\} \\
&= \{\alpha| \exists \alpha \in I \text{ such that } \alpha \in A_\alpha\}\\
\text{intersection: } \bigcap_{\alpha \in I} A_\alpha &= \{\alpha| \alpha \in A_\alpha \text{ for every } \alpha \in I\}\\
&= \{\alpha| \forall \alpha \in I \text{ we have } \alpha \in A_\alpha\}
\end{align*}
\section{Maps}
*Maps in general.* \\
Map = function = mapping = transformation\\
Let $A,B$ be *sets* (+$\emptyset$). A *map* f:
\begin{align*}
A &\rightarrow B\\
a&\mapsto f(a)
\end{align*}
is an *assignment*: to $\forall a \in A$ f assign a unique $b\in B$ which is called the *value of f at a* and is denoted by $f(a)$.\\
*Note:* f assigns to $\forall a \in A$ a value, but there may exists $b \in B$ st $f(a_1)\neq b$ $\forall a \in A$.\\
$A$ is called the *domain* (of definition) of f\\
$B$ is the *target space* of f
